const textarea = document.getElementById("messageInput"); const sendBtn = document.getElementById("sendBtn"); const chatContainer = document.getElementById("chatContainer"); // Invia con Enter textarea.addEventListener("keydown", function(e) { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendBtn.click(); } }); // Simulazione invio messaggio sendBtn.addEventListener("click", () => { const text = textarea.value.trim(); if (text) { const msg = document.createElement("div"); msg.className = "message"; msg.style.alignSelf = "flex-end"; msg.style.width = "60%"; msg.innerHTML = `

${text}

${new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}
`; chatContainer.appendChild(msg); textarea.value = ""; } });
